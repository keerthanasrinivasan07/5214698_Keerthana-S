#include <stdio.h>

int birthday(int s[], int n, int b, int m) {
    int count = 0;
    int i, j;

    for (i = 0; i <= n - m; i++) {
        int sum = 0;
        for (j = i; j < i + m; j++) {
            sum += s[j];
        }
        if (sum == b) {
            count++;
        }
    }
    return count;
}

int main() {
    int n;
    scanf("%d", &n);

    int s[1000];
    int i;

    for (i = 0; i < n; i++) {
        scanf("%d", &s[i]);
    }

    int b, m;
    scanf("%d %d", &b, &m);

    int result = birthday(s, n, b, m);
    printf("%d\n", result);

    return 0;
}
