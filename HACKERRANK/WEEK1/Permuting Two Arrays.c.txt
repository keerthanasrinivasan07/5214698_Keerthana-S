#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n, k;
        scanf("%d%d", &n, &k);

        int A[1000], B[1000];
        int i, j, temp;

        for (i = 0; i < n; i++) scanf("%d", &A[i]);
        for (i = 0; i < n; i++) scanf("%d", &B[i]);

        for (i = 0; i < n - 1; i++) {
            for (j = i + 1; j < n; j++) {
                if (A[i] > A[j]) {
                    temp = A[i];
                    A[i] = A[j];
                    A[j] = temp;
                }
            }
        }

        for (i = 0; i < n - 1; i++) {
            for (j = i + 1; j < n; j++) {
                if (B[i] < B[j]) {
                    temp = B[i];
                    B[i] = B[j];
                    B[j] = temp;
                }
            }
        }

        int possible = 1;
        for (i = 0; i < n; i++) {
            if (A[i] + B[i] < k) {
                possible = 0;
                break;
            }
        }

        if (possible)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
