#include <stdio.h>

int main() {
    int a[5];
    long sum = 0;
    int mini, maxi;

    for (int i = 0; i < 5; i++) {
        scanf("%d", &a[i]);
        sum += a[i];
    }

    mini = maxi = a[0];

    for (int i = 1; i < 5; i++) {
        if (a[i] < mini) mini = a[i];
        if (a[i] > max) max = a[i];
    }

    long miniSum = sum - max;
    long maxSum = sum - mini;

    printf("%ld %ld\n", miniSum, maxSum);
    return 0;
}
