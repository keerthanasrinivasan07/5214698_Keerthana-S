#include <stdio.h>

int main() {
    long long n;
    scanf("%lld", &n);

    if (n == 0) {
        printf("1\n");
        return 0;
    }

    long long t = n;
    int zeros = 0;

    while (t) {
        if ((t & 1) == 0) zeros++;
        t >>= 1;
    }

    long long result = 1;
    for (int i = 0; i < zeros; i++) {
        result <<= 1;
    }

    printf("%lld\n", result);
    return 0;
}
