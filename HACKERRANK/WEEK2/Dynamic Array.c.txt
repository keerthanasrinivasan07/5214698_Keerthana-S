#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

    int **seq = malloc(n * sizeof(int *));
    int *size = malloc(n * sizeof(int));
    int *capacity = malloc(n * sizeof(int));
    int last = 0;

    for (int i = 0; i < n; i++) {
        seq[i] = NULL;
        size[i] = 0;
        capacity[i] = 0;
    }

    for (int i = 0; i < q; i++) {
        int type, x, y;
        scanf("%d %d %d", &type, &x, &y);
        int idx = (x ^ last) % n;

        if (type == 1) {
            if (size[idx] == capacity[idx]) {
                capacity[idx] = capacity[idx] == 0 ? 1 : capacity[idx] * 2;
                seq[idx] = realloc(seq[idx], capacity[idx] * sizeof(int));
            }
            seq[idx][size[idx]] = y;
            size[idx]++;
        } else {
            int pos = y % size[idx];
            last = seq[idx][pos];
            printf("%d\n", last);
        }
    }

    for (int i = 0; i < n; i++) {
        free(seq[i]);
    }
    free(seq);
    free(size);
    free(capacity);

    return 0;
}
