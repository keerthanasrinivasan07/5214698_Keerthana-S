#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int minDiff(int size, int arr[], int n) {
    qsort(arr, n, sizeof(int), cmp);

    int diff = arr[size - 1] - arr[0];
    for (int i = 1; i <= n - size; i++) {
        int curr = arr[i + size - 1] - arr[i];
        if (curr < diff)
            diff = curr;
    }
    return diff;
}

int main() {
    int n, k;
    scanf("%d%d", &n, &k);

    int a[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &a[i]);

    printf("%d\n", minDiff(k, a, n));

    return 0;
}
