#include <stdio.h>
#include <stdlib.h>

int month[] = {31,28,31,30,31,30,31,31,30,31,30,31};

int isLeap(int y) {
    if (y % 400 == 0) return 1;
    if (y % 100 == 0) return 0;
    if (y % 4 == 0) return 1;
    return 0;
}

int main() {
    int d1, m1, y1;
    int d2, m2, y2;
    scanf("%d-%d-%d", &d1, &m1, &y1);
    scanf("%d-%d-%d", &d2, &m2, &y2);

    int count = 0;
    while (1) {
        // Concatenate without leading zeros
        char buf[20];
        sprintf(buf, "%d%d%d", d1, m1, y1);
        long long num = atoll(buf);

        if (num % 4 == 0 || num % 7 == 0) {
            count++;
        }

        // Stop when we've reached the end date
        if (d1 == d2 && m1 == m2 && y1 == y2) {
            break;
        }

        // Increment date
        d1++;
        int maxd = month[m1 - 1];
        if (m1 == 2 && isLeap(y1)) {
            maxd = 29;
        }
        if (d1 > maxd) {
            d1 = 1;
            m1++;
            if (m1 > 12) {
                m1 = 1;
                y1++;
            }
        }
    }

    printf("%d\n", count);
    return 0;
}
