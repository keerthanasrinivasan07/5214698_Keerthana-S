#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[1000], used[1000] = {0};
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int p = 0;
    for(int i = 0; i < n; i++) {
        if(!used[i]) {
            for(int j = i + 1; j < n; j++) {
                if(!used[j] && a[i] == a[j]) {
                    p++;
                    used[i] = used[j] = 1;
                    break;
                }
            }
        }
    }

    printf("%d\n", p);
    return 0;
}
