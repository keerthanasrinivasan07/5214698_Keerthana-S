#include <stdio.h>
#include <string.h>

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n;
        scanf("%d", &n);

        char grid[100][100];
        for (int i = 0; i < n; i++) {
            scanf("%s", grid[i]);
        }

        // sort each row using bubble sort
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - 1; j++) {
                for (int k = 0; k < n - j - 1; k++) {
                    if (grid[i][k] > grid[i][k + 1]) {
                        char temp = grid[i][k];
                        grid[i][k] = grid[i][k + 1];
                        grid[i][k + 1] = temp;
                    }
                }
            }
        }

        // check columns
        int ok = 1;
        for (int col = 0; col < n && ok; col++) {
            for (int row = 0; row < n - 1; row++) {
                if (grid[row][col] > grid[row + 1][col]) {
                    ok = 0;
                    break;
                }
            }
        }

        if (ok)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
