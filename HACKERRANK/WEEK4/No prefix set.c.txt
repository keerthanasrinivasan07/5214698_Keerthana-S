#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Node {
    struct Node *next[26];
    int end;
};

struct Node* getNode() {
    struct Node *p = (struct Node*)malloc(sizeof(struct Node));
    for (int i = 0; i < 26; i++) p->next[i] = NULL;
    p->end = 0;
    return p;
}

int main() {
    int n;
    scanf("%d", &n);

    struct Node *root = getNode();
    char word[101];
    for (int i = 0; i < n; i++) {
        scanf("%s", word);
        struct Node *curr = root;
        int bad = 0;
        for (int j = 0; word[j]; j++) {
            int k = word[j] - 'a';
            if (curr->next[k] == NULL)
                curr->next[k] = getNode();
            curr = curr->next[k];

            if (curr->end) {  // earlier word ended here
                bad = 1;
                break;
            }
        }
        if (!bad) {
            for (int x = 0; x < 26; x++) {
                if (curr->next[x] != NULL) {
                    bad = 1;
                    break;
                }
            }
        }
        if (bad) {
            printf("BAD SET\n%s\n", word);
            return 0;
        }
        curr->end = 1;
    }
    printf("GOOD SET\n");
    return 0;
}
