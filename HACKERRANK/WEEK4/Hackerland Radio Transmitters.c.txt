#include <stdio.h>
#include <stdlib.h>

int c(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int main() {
    int n,k;
    scanf("%d %d",&n,&k);
    int x[n];
    for(int i=0;i<n;i++) scanf("%d",&x[i]);
    qsort(x,n,sizeof(int),c);

    int ans=0,i=0;
    while(i<n){
        ans++;
        int loc=x[i]+k;
        while(i<n && x[i]<=loc) i++;
        i--;
        loc=x[i]+k;
        while(i<n && x[i]<=loc) i++;
    }
    printf("%d",ans);
    return 0;
}
