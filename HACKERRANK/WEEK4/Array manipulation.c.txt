#include <stdio.h>

long long a[10000010]; // big enough for constraints

int main() {
    int n,m;
    scanf("%d %d",&n,&m);
    for(int i=0;i<m;i++){
        int l,r; long long k;
        scanf("%d %d %lld",&l,&r,&k);
        a[l]+=k;
        if(r+1<=n) a[r+1]-=k; // safe check
    }
    long long s=0,mx=0;
    for(int i=1;i<=n;i++){
        s+=a[i];
        if(s>mx) mx=s;
    }
    printf("%lld\n",mx);
    return 0;
}
