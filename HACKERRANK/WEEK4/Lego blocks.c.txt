#include <stdio.h>
#define MOD 1000000007

int main() {
    int t;
    scanf("%d", &t);

    while(t--) {
        int n, m;
        scanf("%d %d", &n, &m);

        long row[m+1];
        long total[m+1];
        long solid[m+1];

        row[0] = 1;
        for(int i=1; i<=m; i++) {
            row[i] = 0;
            for(int j=1; j<=4; j++) {
                if(i-j >= 0) row[i] = (row[i] + row[i-j]) % MOD;
            }
        }

        for(int i=1; i<=m; i++) {
            long r = 1;
            for(int k=0; k<n; k++) {
                r = (r * row[i]) % MOD;
            }
            total[i] = r;
        }

        solid[0] = 0;
        for(int i=1; i<=m; i++) {
            solid[i] = total[i];
            for(int j=1; j<i; j++) {
                solid[i] = (solid[i] - (solid[j] * total[i-j]) % MOD + MOD) % MOD;
            }
        }

        printf("%ld\n", solid[m]);
    }

    return 0;
}
