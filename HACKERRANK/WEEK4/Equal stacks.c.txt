#include <stdio.h>

int main() {
    int n1, n2, n3;
    scanf("%d %d %d", &n1, &n2, &n3);

    int a[n1], b[n2], c[n3];
    int i;

    for(i=0; i<n1; i++) scanf("%d", &a[i]);
    for(i=0; i<n2; i++) scanf("%d", &b[i]);
    for(i=0; i<n3; i++) scanf("%d", &c[i]);

    int s1=0, s2=0, s3=0;
    for(i=0; i<n1; i++) s1 += a[i];
    for(i=0; i<n2; i++) s2 += b[i];
    for(i=0; i<n3; i++) s3 += c[i];

    int p1=0, p2=0, p3=0;   // pointers to current top

    while(1) {
        if(s1==s2 && s2==s3) {
            printf("%d", s1);
            return 0;
        }
        if(p1==n1 || p2==n2 || p3==n3) {
            printf("0");
            return 0;
        }
        if(s1 >= s2 && s1 >= s3) {
            s1 -= a[p1];
            p1++;
        }
        else if(s2 >= s1 && s2 >= s3) {
            s2 -= b[p2];
            p2++;
        }
        else {
            s3 -= c[p3];
            p3++;
        }
    }
}
