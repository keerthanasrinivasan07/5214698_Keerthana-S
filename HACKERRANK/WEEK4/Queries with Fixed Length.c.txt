#include <stdio.h>
#include <limits.h>

int a[100005], d[100005];

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

    for(int i=0;i<n;i++) scanf("%d",&a[i]);

    while(q--) {
        int k;
        scanf("%d",&k);

        int f=0,b=-1,ans=INT_MAX;
        for(int i=0;i<n;i++) {
            while(f<=b && a[d[b]]<=a[i]) b--;
            d[++b]=i;
            if(d[f]<=i-k) f++;
            if(i>=k-1) {
                if(a[d[f]]<ans) ans=a[d[f]];
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
