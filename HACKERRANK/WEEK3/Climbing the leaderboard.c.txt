#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int ranked[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &ranked[i]);
    }

    int m;
    scanf("%d", &m);
    int player[m];
    for (int i = 0; i < m; i++) {
        scanf("%d", &player[i]);
    }

    // remove duplicates from ranked
    int uniq[n];
    int ucount = 0;
    uniq[ucount++] = ranked[0];
    for (int i = 1; i < n; i++) {
        if (ranked[i] != ranked[i - 1]) {
            uniq[ucount++] = ranked[i];
        }
    }

    int pos = ucount - 1; // start from the end of leaderboard
    for (int i = 0; i < m; i++) {
        while (pos >= 0 && player[i] >= uniq[pos]) {
            pos--;
        }
        printf("%d\n", pos + 2);
    }

    return 0;
}
