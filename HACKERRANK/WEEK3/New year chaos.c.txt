#include <stdio.h>

int main(){
int t;scanf("%d",&t);
while(t--){
 int n;scanf("%d",&n);
 int q[n];
 for(int i=0;i<n;i++) scanf("%d",&q[i]);
 int ans=0,flag=0;
 for(int i=0;i<n;i++){
   if(q[i]-(i+1)>2){printf("Too chaotic\n");flag=1;break;}
 }
 if(flag) continue;
 for(int i=0;i<n;i++){
   int k=q[i]-2;
   if(k<0) k=0;
   for(int p=k;p<i;p++){
     if(q[p]>q[i]) ans++;
   }
 }
 printf("%d\n",ans);
}
return 0;
}
