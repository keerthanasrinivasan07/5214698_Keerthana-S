#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char* b(char *s){
    int n=strlen(s);
    char st[1000];
    int top=-1;
    for(int i=0;i<n;i++){
        if(s[i]=='('||s[i]=='{'||s[i]=='['){
            st[++top]=s[i];
        }else{
            if(top==-1){
                return "NO";
            }
            char c=st[top--];
            if((s[i]==')'&&c!='(')||(s[i]=='}'&&c!='{')||(s[i]==']'&&c!='[')){
                return "NO";
            }
        }
    }
    if(top==-1) return "YES";
    else return "NO";
}

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        char str[1000];
        scanf("%s",str);
        printf("%s\n",b(str));
    }
    return 0;
}
